```.py
def dictcounter(dicc,msg):
    for i in dicc:
        if i in msg:
            dicc[i] += 1
    print(dicc)
dictcounter({'w': 0, 'l': 0, 'c': 0}, "hello world")
dictcounter({'a': 0, 'e': 0, 'i': 0, 'o': 0, 'u': 0}, "Why did I choose CS?")
```
